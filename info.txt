SELECT g.id, g.business_id, g.nombre, b.name AS biz_name
FROM agrupaciones g
JOIN app_businesses b ON b.id = g.business_id
WHERE g.business_id IN (
  SELECT id FROM app_businesses
  WHERE organization_id = (
    SELECT organization_id FROM app_businesses WHERE id = 175
  )
)
ORDER BY g.business_id, g.id;



SELECT 
  g.id,
  g.nombre,
  g.business_id,
  jsonb_array_length(COALESCE(g.articulos, '[]'::jsonb)) AS cant_articulos,
  (
    SELECT jsonb_agg(DISTINCT art->>'subrubro')
    FROM jsonb_array_elements(COALESCE(g.articulos, '[]'::jsonb)) AS art
    WHERE art->>'subrubro' IS NOT NULL
  ) AS subrubros_distintos
FROM agrupaciones g
WHERE g.business_id = 175
ORDER BY g.id;
